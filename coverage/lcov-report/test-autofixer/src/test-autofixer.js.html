const fs = require('fs/promises');
const path = require('path');
const { runCLI } = require('jest');
const glob = require('glob');
 
class TestAutoFixer {
  constructor(options = {}) {
    this.options = {
      testPath: options.testPath || '.',
      filePattern: options.filePattern || '**/*.test.js',
      dryRun: options.dryRun || false,
      interactive: options.interactive || false,
      logFile: options.logFile || 'test-autofixer.log'
    };
    this.logger = null;
  }
 
  async initializeLogger() {
    try {
      await fs.writeFile(this.options.logFile, ''); // Clear/create the log file
    } catch (error) {
      console.error('Error initializing logger:', error);
    }
  }
 
  async log(message) {
    try {
      await fs.appendFile(this.options.logFile, message + '\n');
    } catch (error) {
      console.error('Error writing to log:', error);
    }
  }
 
  async finalizeLogger() {
    // No need to close anything when using fs.writeFile/appendFile
  }
 
  async findTestFiles() {
    return new Promise((resolve, reject) => {
      const pattern = path.join(this.options.testPath, this.options.filePattern);
      glob(pattern, (err, files) => {
        if (err) reject(err);
        else resolve(files);
      });
    });
  }
 
  extractExpectedValue(failureMessage) {
    const match = failureMessage.match(/Expected:\s*("[^"]+"|'[^']+')/);
    return match ? match[1] : null;
  }
 
  extractReceivedValue(failureMessage) {
    const match = failureMessage.match(/Received:\s*("[^"]+"|'[^']+')/);
    return match ? match[1] : null;
  }
 
  async generateFix(test) {
    return {
      testName: test.name,
      oldValue: test.expected,
      newValue: test.received,
      fixed: true
    };
  }
 
  async runTests() {
    console.log('Running tests...');
    try {
      const jestConfig = {
        rootDir: process.cwd(),
        testEnvironment: 'node',
        verbose: true,
        testMatch: [
          "**/tests/**/*.test.js",
          "**/__tests__/**/*.test.js"
        ],
        testPathIgnorePatterns: [
          "/node_modules/",
          "/test-autofixer/src/__tests__/fixtures/"
        ]
      };
 
      const { results } = await runCLI({
        _: [],
        $0: 'jest',
        config: jestConfig,
        passWithNoTests: false,
        runInBand: true,
        silent: false
      }, [process.cwd()]);
 
      return {
        success: results.success,
        testResults: results.testResults || []
      };
    } catch (error) {
      console.error('Error running tests:', error);
      return {
        success: false,
        testResults: [],
        error: error.message
      };
    }
  }
 
  async fixTest(testFile, mismatchDetails) {
    console.log(`\nFixing test in ${testFile}`);
    console.log('Mismatch details:', mismatchDetails);
    
    try {
      let content = await fs.readFile(testFile, 'utf8');
      let fixed = false;
 
      if (mismatchDetails.expected && mismatchDetails.received) {
        const patterns = [
          // Jest expect patterns with any quote type
          {
            pattern: new RegExp('expect\\([^)]+\\)\\.toBe\\([\'"`]' + this.escapeRegExp(mismatchDetails.expected) + '[\'"`]\\)'),
            replacement: m => m.group().replace(mismatchDetails.expected, mismatchDetails.received)
          },
          {
            pattern: new RegExp('expect\\([^)]+\\)\\.toEqual\\([\'"`]' + this.escapeRegExp(mismatchDetails.expected) + '[\'"`]\\)'),
            replacement: m => m.group().replace(mismatchDetails.expected, mismatchDetails.received)
          },
          {
            pattern: new RegExp('expect\\([^)]+\\)\\.toContain\\([\'"`]' + this.escapeRegExp(mismatchDetails.expected) + '[\'"`]\\)'),
            replacement: m => m.group().replace(mismatchDetails.expected, mismatchDetails.received)
          },
          
          // Error message patterns
          {
            pattern: new RegExp('ValidationError:\\s*[\'"`]([^\'"`]+)[\'"`]'),
            replacement: m => `ValidationError: "${mismatchDetails.received}"`
          },
          {
            pattern: new RegExp('error\\.message\\s*===?\\s*[\'"`]([^\'"`]+)[\'"`]'),
            replacement: m => `error.message === "${mismatchDetails.received}"`
          },
          {
            pattern: new RegExp('error\\.errors\\.includes\\([\'"`]([^\'"`]+)[\'"`]\\)'),
            replacement: m => `error.errors.includes("${mismatchDetails.received}")`
          }
        ];
 
        for (const {pattern, replacement} of patterns) {
          if (pattern.test(content)) {
            content = content.replace(pattern, replacement);
            fixed = true;
          }
        }
      }
 
      if (fixed) {
        await fs.writeFile(testFile, content, 'utf8');
        console.log(`✓ Fixed test in ${testFile}`);
        return true;

      console.log(`✗ Could not fix test in ${testFile}`);
      return false;

      return false;

  }
 
  escapeRegExp(string) {
    return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  }
 
  async run() {
    try {
      // First test run
      console.log('Running initial tests...');
      const initialResults = await this.runTests();
 
      if (initialResults.success) {
        console.log('All tests passed! No fixes needed.');
        return;
      }
 
      // Process failing tests
      let fixedAny = false;
      for (const result of initialResults.testResults) {
        if (!result.testResults) continue;
 
        const failingTests = result.testResults.filter(t => t.status === 'failed');
        for (const test of failingTests) {
          console.log(`\nProcessing failing test: ${test.fullName}`);
          
          for (const failureMessage of test.failureMessages || []) {
            const mismatchDetails = {
              expected: this.extractExpectedValue(failureMessage),
              received: this.extractReceivedValue(failureMessage)
            };
            if (mismatchDetails.expected && mismatchDetails.received) {
              const fixed = await this.fixTest(result.testFilePath, mismatchDetails);
              if (fixed) fixedAny = true;
            }
          }
        }
      }
 
      if (!fixedAny) {
        console.log('Could not automatically fix any tests. Manual intervention needed.');
        return;
      }
 
      // Re-run tests to verify fixes
      console.log('\nRe-running tests to verify fixes...');
      const finalResults = await this.runTests();
 
      if (finalResults.success) {
        console.log('All tests passed after fixing!');
      } else {
        console.log('Some tests are still failing. Manual intervention might be needed.');
      }
    } catch (error) {
      console.error('Error running test-autofixer:', error);
      throw error;
    }
  }
}
